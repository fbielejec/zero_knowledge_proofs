# computes e(Q \in G2, P \in G1)
def pairing(P,Q):
  ePQ = bn128.pairing( (bn128.FQ2([int(Q[0][0]), int(Q[0][1])]),
                        bn128.FQ2([int(Q[1][0]), int(Q[1][1])])),
                       (bn128.FQ(int(P[0])), bn128.FQ(int(P[1]))) );
  # returns an element in FQ12 (extension tower Fp^2 -> Fp^6 -> Fp12)
  return ePQ

# proof : a*b = c
a=3; b=4; c=12;
aG1 = a * G1
bG2 = b * G2
cG1 = 12 * G1

lhs = pairing(aG1,bG2)
rhs = pairing(cG1, G2)
lhs == rhs

# proof : a*b + c*d = 24
a=2;b=3;c=2;d=9
